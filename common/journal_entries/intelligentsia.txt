#Intelligentsia in Government
je_intelligentsia_1 = {
	group = je_group_internal_affairs
	
	progressbar = yes
	current_value = {
		value = root.var:je_i_1_var
	}
	
	possible = {
		ig:ig_intelligentsia = {
			is_marginal = no
			is_in_government = yes
		}
	}

	is_shown_when_inactive = {
		NOT = { has_variable = lintelligentsia_1_done }
		has_technology_researched = democracy
	}

	immediate = {
		set_variable = {
			name = je_i_1_var
			value = 0
		}
	}

	goal_add_value = {
		add = 12
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					government_legitimacy >= 50
					ig:ig_intelligentsia = {
						is_in_government = yes	
					}
					NOT = {
						any_interest_group = {
							is_insurrectionary = yes
						}
					}
				}
				change_variable = {
					name = je_i_1_var
					add = 1
				}
			}
			else = {
				set_variable = {
					name = je_i_1_var
					value = 0
				}	
			}	
		}
	}

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

	on_complete = {
		trigger_event = { id = intelligentsia.1 }
		set_variable = intelligentsia_1_done
	}
}

			set_variable = intelligentsia_1_done

		}
}
